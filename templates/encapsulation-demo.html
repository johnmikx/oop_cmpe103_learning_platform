üìù How Encapsulation Works

class BankAccount: def __init__(self, initial_balance=0): # Private attributes (name mangling with __) self.__balance = initial_balance # üîí PRIVATE self.__transactions = [] # üîí PRIVATE self.__account_number = "ACC123" # üîí PRIVATE # üîì PUBLIC methods for controlled access def get_balance(self): """Safe way to access balance""" return self.__balance def deposit(self, amount): """Safe way to add money with validation""" if amount <= 0: raise ValueError("Amount must be positive") self.__balance += amount self.__transactions.append(f"Deposit: +${amount}") return True def withdraw(self, amount): """Safe way to remove money with validation""" if amount <= 0: raise ValueError("Amount must be positive") if amount > self.__balance: raise ValueError("Insufficient funds") self.__balance -= amount self.__transactions.append(f"Withdrawal: -${amount}") return True # üîí PRIVATE method (internal use only) def __validate_transaction(self, amount): return amount > 0 and amount <= 10000
